# Router: /orders (ONLINE)

import uuid
from datetime import datetime, timedelta, timezone

from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session

from app.core.db import get_db
from app.core.auth_dep import get_current_user
from app.models.order import Order, OrderChannel, OrderStatus
from app.models.allocation import Allocation, AllocationState
from app.services import backend_client
from app.schemas.orders import CreateOrderIn, OrderOut

router = APIRouter(prefix="/orders", tags=["orders"])

ALLOC_TTL_SEC = 120  # 2 minutos (reserva curta enquanto paga)
PICKUP_WINDOW_HOURS = 2  # 2 horas (retirada)


def _utcnow() -> datetime:
    return datetime.now(timezone.utc)


def _to_iso(dt: datetime) -> str:
    # backend espera ISO string; manter timezone explícito
    return dt.isoformat()


def _ensure_amount_cents(pricing: dict) -> int:
    amount_cents = pricing.get("amount_cents") or pricing.get("price_cents")
    if amount_cents is None:
        raise HTTPException(
            status_code=502,
            detail="pricing missing amount_cents/price_cents from backend",
        )
    try:
        return int(amount_cents)
    except Exception:
        raise HTTPException(status_code=502, detail="pricing amount_cents inválido")


def _get_order_or_404(db: Session, order_id: str) -> Order:
    order = db.query(Order).filter(Order.id == order_id).first()
    if not order:
        raise HTTPException(status_code=404, detail="order not found")
    return order


@router.post("", response_model=OrderOut, status_code=status.HTTP_201_CREATED)
def create_order(
    payload: CreateOrderIn,
    db: Session = Depends(get_db),
    user=Depends(get_current_user),
):
    """
    ONLINE: cria Order em PAYMENT_PENDING + Allocation RESERVED_PENDING_PAYMENT

    1) Consulta preço no backend (fonte de verdade)
    2) locker_allocate no backend (reserva curta 2 min)
    3) Persiste Order + Allocation local
    """

    # 1) Preço no backend
    pricing = backend_client.get_sku_pricing(payload.region, payload.sku_id)
    amount_cents = _ensure_amount_cents(pricing)

    # 2) Allocate (reserva curta)
    request_id = str(uuid.uuid4())
    alloc = backend_client.locker_allocate(
        payload.region,
        payload.sku_id,
        ALLOC_TTL_SEC,
        request_id,
    )

    allocation_id = alloc.get("allocation_id")
    slot = alloc.get("slot")
    ttl_sec = int(alloc.get("ttl_sec", ALLOC_TTL_SEC))

    if not allocation_id or slot is None:
        raise HTTPException(status_code=502, detail="locker allocate missing allocation_id/slot")

    # 3) Persist local
    order = Order(
        id=str(uuid.uuid4()),
        user_id=user.id,
        channel=OrderChannel.ONLINE,
        region=payload.region,
        totem_id=payload.totem_id,
        sku_id=payload.sku_id,
        amount_cents=amount_cents,
        status=OrderStatus.PAYMENT_PENDING,
    )
    db.add(order)
    db.flush()

    allocation = Allocation(
        id=str(allocation_id),
        order_id=order.id,
        slot=int(slot),
        state=AllocationState.RESERVED_PENDING_PAYMENT,
        locked_until=None,
    )
    db.add(allocation)
    db.commit()

    return OrderOut(
        order_id=order.id,
        channel=order.channel.value,
        status=order.status.value,
        amount_cents=order.amount_cents,
        allocation={"allocation_id": allocation.id, "slot": allocation.slot, "ttl_sec": ttl_sec},
    )


@router.get("/{order_id}", response_model=OrderOut)
def get_order(
    order_id: str,
    db: Session = Depends(get_db),
    user=Depends(get_current_user),
):
    """
    Retorna o status do pedido ONLINE e dados de allocation (slot/locked_until/state)
    """
    order = _get_order_or_404(db, order_id)

    # segurança: só dono
    if str(order.user_id) != str(user.id):
        raise HTTPException(status_code=403, detail="forbidden")

    allocation = db.query(Allocation).filter(Allocation.order_id == order.id).first()
    alloc_out = None
    if allocation:
        alloc_out = {
            "allocation_id": allocation.id,
            "slot": allocation.slot,
            "locked_until": allocation.locked_until.isoformat() if allocation.locked_until else None,
            "state": allocation.state.value if hasattr(allocation.state, "value") else str(allocation.state),
        }

    return OrderOut(
        order_id=order.id,
        channel=order.channel.value,
        status=order.status.value,
        amount_cents=order.amount_cents,
        allocation=alloc_out,
    )


@router.post("/{order_id}/payment/approved", response_model=OrderOut)
def payment_approved(
    order_id: str,
    db: Session = Depends(get_db),
    user=Depends(get_current_user),
):
    """
    Chamado após pagamento aprovado.
    Ação:
    - Order: PAYMENT_PENDING -> PAID
    - Backend: locker_commit(allocation_id, locked_until=+2h)
    - Allocation: LOCKED_FOR_PICKUP + locked_until local
    """
    order = _get_order_or_404(db, order_id)

    # segurança: só dono
    if str(order.user_id) != str(user.id):
        raise HTTPException(status_code=403, detail="forbidden")

    allocation = db.query(Allocation).filter(Allocation.order_id == order.id).first()
    if not allocation:
        raise HTTPException(status_code=409, detail="order has no allocation")

    if order.status not in (OrderStatus.PAYMENT_PENDING, OrderStatus.CREATED):
        raise HTTPException(status_code=409, detail=f"invalid state transition from {order.status.value}")

    locked_until = _utcnow() + timedelta(hours=PICKUP_WINDOW_HOURS)

    # 🔒 Backend: “commit” com lock longo 2h
    try:
        backend_client.locker_commit(order.region, allocation.id, _to_iso(locked_until))
    except Exception as e:
        raise HTTPException(status_code=502, detail=f"failed to commit allocation on backend: {e}")

    # Persist local
    order.status = OrderStatus.PAID
    allocation.state = AllocationState.LOCKED_FOR_PICKUP
    allocation.locked_until = locked_until
    db.commit()

    return OrderOut(
        order_id=order.id,
        channel=order.channel.value,
        status=order.status.value,
        amount_cents=order.amount_cents,
        allocation={
            "allocation_id": allocation.id,
            "slot": allocation.slot,
            "locked_until": allocation.locked_until.isoformat(),
            "state": allocation.state.value if hasattr(allocation.state, "value") else str(allocation.state),
        },
    )


@router.post("/{order_id}/cancel", response_model=OrderOut)
def cancel_order(
    order_id: str,
    db: Session = Depends(get_db),
    user=Depends(get_current_user),
):
    """
    Cancelamento ONLINE (antes do pickup):
    - Backend: locker_release(allocation_id) (best-effort)
    - Local: Order -> CANCELLED + Allocation -> RELEASED
    """
    order = _get_order_or_404(db, order_id)

    if str(order.user_id) != str(user.id):
        raise HTTPException(status_code=403, detail="forbidden")

    allocation = db.query(Allocation).filter(Allocation.order_id == order.id).first()
    if not allocation:
        raise HTTPException(status_code=409, detail="order has no allocation")

    if order.status in (OrderStatus.DISPENSED, OrderStatus.CANCELLED, OrderStatus.EXPIRED):
        raise HTTPException(status_code=409, detail=f"cannot cancel from {order.status.value}")

    # backend: liberar (best-effort)
    try:
        backend_client.locker_release(order.region, allocation.id)
    except Exception:
        pass

    order.status = OrderStatus.CANCELLED
    allocation.state = AllocationState.RELEASED
    allocation.locked_until = None
    db.commit()

    return OrderOut(
        order_id=order.id,
        channel=order.channel.value,
        status=order.status.value,
        amount_cents=order.amount_cents,
        allocation={"allocation_id": allocation.id, "slot": allocation.slot},
    )
